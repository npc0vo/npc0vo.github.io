def rc4_crypt(key, data):
    S = list(range(255,-1,-1))
    j = 0
    out = []

    # 初始化S盒
    for i in range(256):
        j = (j + S[i] + ord(key[i % len(key)]) ) % 256
        S[i], S[j] = S[j], S[i]

    # 生成密钥流并加解密
    i = j = 0
    for char in data:
        i = (i + 1) % 256
        j = (j + S[i]) % 256
        S[i], S[j] = S[j], S[i]
        out.append(((char) ^ S[(S[i] + S[j]) % 256]^0x2f))

    return out
